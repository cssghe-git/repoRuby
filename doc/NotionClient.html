def update_page_date_envoi(page_id, iso_datetime)
    body = {
        properties: {
            "Date envoi" => {
                date: { start: iso_datetime }
            }
        }
    }
    self.class.patch("/pages/#{page_id}", headers: @headers, body: body.to_json)
end
